                    class Vision:
                    # given a list of [x, y, w, h] rectangles returned 
                    # by find(), convert those into a list of
                    # [x, y] positions in the center of those 
                    #  rectangles where we can click on those found items
                    def get_click_points(self, rectangles):
                        points = []
                        
                        # Loop over all the rectangles
                        for (x, y, w, h) in rectangles:
                            # Determine the center position
                            center_x = x + int(w/2)
                            center_y = y + int(h/2)
                            # Save the points
                            points.append((center_x, center_y))
                 
                        return points
